- [ ] **Step 3: Implement two-pass walk in `find_repo_root`**

```python
def find_repo_root(file_path: Path) -> Path | None:
    """Walk up from file_path looking for a repo-root marker.

    Two-pass strategy:
      Pass 1: walk up to 32 levels collecting any ancestor that has a
              `.chameleon/` directory. If at least one is found, return
              the DEEPEST one (the closest committed profile wins, so a
              workspace sub-profile takes priority over the root).
      Pass 2: no `.chameleon/` found anywhere. Walk up again; return the
              FIRST ancestor with any non-`.chameleon` marker
              (.git, package.json, tsconfig.json, Gemfile, pyproject.toml,
              go.mod, Cargo.toml).

    BUG-NEW-002 (v0.5.7): the previous one-pass walk stopped at the first
    ancestor with any marker, so monorepos with .chameleon at the root and
    package.json at each workspace returned the workspace as repo_root and
    masked the root profile.
    """
    current = file_path.expanduser()
    if current.is_file():
        current = current.parent
    try:
        current = current.resolve()
    except OSError:
        return None

    # Pass 1: deepest .chameleon ancestor
    candidates: list[Path] = []
    walker = current
    for _ in range(32):
        if (walker / ".chameleon").exists():
            candidates.append(walker)
        parent = walker.parent
        if parent == walker:
            break
        walker = parent
    if candidates:
        return candidates[0]  # deepest (first encountered going up)

    # Pass 2: first ancestor with any other marker
    walker = current
    for _ in range(32):
        for marker in REPO_ROOT_MARKERS:
            if marker == ".chameleon":
                continue
            if (walker / marker).exists():
                return walker
        parent = walker.parent
        if parent == walker:
            break
        walker = parent
    return None
```

```python
def _parse_rubocop_yaml(path: Path) -> tuple[dict | None, str | None]:
    try:
        import yaml
    except ImportError:
        return None, "PyYAML unavailable; cannot parse rubocop config"
    try:
        text = path.read_text(encoding="utf-8", errors="replace")
        parsed = yaml.safe_load(text)
        if not isinstance(parsed, dict):
            return None, f"rubocop config is not a YAML mapping: {type(parsed).__name__}"
        # Strip `inherit_from` reference — we don't follow it, the cops listed
        # in the file directly are still useful guidance.
        return parsed, None
    except yaml.YAMLError as e:
        return None, f"YAML parse error: {e}"
    except Exception as e:
        return None, f"unexpected error: {e}"
```
